// Rumus Inti: Menghitung Rata-rata Nilai Mahasiswa
const rataRata = totalNilai / jumlahMataKuliah;

// Cek apakah nilai memenuhi Target CPL (misal: 75)
if (rataRata >= targetCPL) {

    statusCPL: status
